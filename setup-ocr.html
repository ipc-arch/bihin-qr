📸 OCR型番自動読み取りガイド

Google Apps Script + Cloud Vision API で備品シールからテキストを自動抽出

やること: iPadで備品シールを撮影 → Driveに保存 → 型番・シリアル番号・管理番号が自動でスプレッドシートに転記される
コスト: 月1,000回まで無料（Cloud Vision API無料枠）
所要時間: 初期セットアップ 20〜30分

全体フロー

📸 iPadで撮影
→
📁 Drive保存
→
🤖 GAS → Cloud Vision
→
📝 型番・S/N自動転記

AppSheetで撮影した写真URLがシートに入る → GASが検知 → OCR → 結果を書き戻し

Step 1: GCPプロジェクトでCloud Vision APIを有効化

  1. console.cloud.google.com に職場の管理者アカウントでログイン
  2. 上部の「プロジェクトを選択」→「新しいプロジェクト」
    プロジェクト名: bihin-ocr（何でもOK）
    「作成」をクリック
  3. 作成したプロジェクトを選択した状態で、左メニュー「APIとサービス」→「ライブラリ」
  4. 検索欄に Cloud Vision API と入力 → 表示されたカードをクリック → 「有効にする」
  5. 左メニュー「APIとサービス」→「認証情報」→「+ 認証情報を作成」→「APIキー」
  6. 生成されたAPIキーをコピーして安全な場所にメモ（後でGASに貼る）
⚠️ APIキーの制限を必ず設定してください。
認証情報ページでAPIキーをクリック → 「APIの制限」→「キーを制限」→ Cloud Vision API のみ選択 → 保存。
これをしないとキーが漏れた場合に他のAPIも呼ばれてしまいます。
💡 大学のGCP利用ポリシーを確認してください。組織によってはプロジェクト作成にIT部門の承認が必要な場合があります。自分がシステム課なら話は早い。

Step 2: Google Apps Scriptのセットアップ

  1. 備品台帳のスプレッドシートを開く
  2. メニューバー「拡張機能」→「Apps Script」
  3. エディタが開いたら、既存のコード（function myFunction(){}）を全削除
  4. 以下のコードを全文コピーして貼り付け

GASコード（全文コピペ）

// ============================================================
// 備品管理 OCR 自動読み取りスクリプト
// Cloud Vision API (TEXT_DETECTION) を使用
// ============================================================

// ★★★ ここにStep 1で取得したAPIキーを貼り付け ★★★
const VISION_API_KEY = 'YOUR_API_KEY_HERE';

// ---------- 設定 ----------
const CONFIG = {
  sheetName: 'Items',          // シート名
  col: {
    photoUrl:    14,  // N列: 型番写真のURL
    ocrRaw:      25,  // Y列: OCR生テキスト（デバッグ用）
    modelParsed: 4,   // D列: 型番（自動転記先）
    serialParsed: 16, // P列: シリアル番号（自動転記先）
    ocrStatus:   26,  // Z列: OCR処理ステータス
  },
  headerRow: 1,
  startRow: 2,
};

// ---------- メイン: 全行スキャン ----------
function runOcrAll() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CONFIG.sheetName);
  if (!sheet) { Logger.log('シート未検出: ' + CONFIG.sheetName); return; }

  const lastRow = sheet.getLastRow();
  if (lastRow < CONFIG.startRow) return;

  let processed = 0;
  for (let row = CONFIG.startRow; row <= lastRow; row++) {
    const photoUrl = sheet.getRange(row, CONFIG.col.photoUrl).getValue();
    const status   = sheet.getRange(row, CONFIG.col.ocrStatus).getValue();

    // 写真URLがあり、まだOCR未実行の行だけ処理
    if (photoUrl && !status) {
      try {
        const result = ocrFromDriveUrl(photoUrl);
        if (result.text) {
          sheet.getRange(row, CONFIG.col.ocrRaw).setValue(result.text);

          const parsed = parseOcrText(result.text);
          // 既に手入力があれば上書きしない
          if (!sheet.getRange(row, CONFIG.col.modelParsed).getValue() && parsed.model) {
            sheet.getRange(row, CONFIG.col.modelParsed).setValue(parsed.model);
          }
          if (!sheet.getRange(row, CONFIG.col.serialParsed).getValue() && parsed.serial) {
            sheet.getRange(row, CONFIG.col.serialParsed).setValue(parsed.serial);
          }
          sheet.getRange(row, CONFIG.col.ocrStatus).setValue('✅ ' + new Date().toLocaleString('ja-JP'));
          processed++;
        } else {
          sheet.getRange(row, CONFIG.col.ocrStatus).setValue('⚠️ テキスト検出なし');
        }
      } catch (e) {
        sheet.getRange(row, CONFIG.col.ocrStatus).setValue('❌ ' + e.message);
        Logger.log('Row ' + row + ' error: ' + e.message);
      }
    }
  }
  Logger.log('OCR完了: ' + processed + '件処理');
}

// ---------- Drive URL → Cloud Vision OCR ----------
function ocrFromDriveUrl(url) {
  // Google Driveの画像URLからファイルIDを抽出
  const fileId = extractDriveFileId(url);
  if (!fileId) {
    throw new Error('DriveファイルID取得不可: ' + url);
  }

  // Drive APIで画像のBase64を取得
  const file = DriveApp.getFileById(fileId);
  const blob = file.getBlob();
  const base64 = Utilities.base64Encode(blob.getBytes());

  // Cloud Vision API 呼び出し
  const visionUrl = 'https://vision.googleapis.com/v1/images:annotate?key=' + VISION_API_KEY;
  const payload = {
    requests: [{
      image: { content: base64 },
      features: [{ type: 'TEXT_DETECTION', maxResults: 1 }]
    }]
  };

  const response = UrlFetchApp.fetch(visionUrl, {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  });

  const json = JSON.parse(response.getContentText());

  if (json.error) {
    throw new Error('Vision API: ' + json.error.message);
  }

  const annotations = json.responses[0].textAnnotations;
  if (!annotations || annotations.length === 0) {
    return { text: '', lines: [] };
  }

  const fullText = annotations[0].description;
  const lines = fullText.split('\n').map(l => l.trim()).filter(l => l);

  return { text: fullText, lines: lines };
}

// ---------- Drive URL → ファイルID ----------
function extractDriveFileId(url) {
  // パターン1: https://drive.google.com/file/d/XXXXX/view
  let m = url.match(/\/file\/d\/([^\/]+)/);
  if (m) return m[1];

  // パターン2: https://drive.google.com/open?id=XXXXX
  m = url.match(/[?&]id=([^&]+)/);
  if (m) return m[1];

  // パターン3: https://lh3.googleusercontent.com/d/XXXXX (AppSheet形式)
  m = url.match(/googleusercontent\.com\/d\/([^?\/]+)/);
  if (m) return m[1];

  // パターン4: AppSheetがDriveに保存した場合のフルパス
  // _RowNumber形式のファイル名が入ることがある
  // その場合はURL自体がファイルIDの可能性
  if (/^[a-zA-Z0-9_-]{20,}$/.test(url)) return url;

  return null;
}

// ---------- OCRテキストから型番・シリアル番号を推定 ----------
function parseOcrText(text) {
  const result = { model: '', serial: '' };
  const lines = text.split('\n').map(l => l.trim()).filter(l => l);

  for (const line of lines) {
    const upper = line.toUpperCase();

    // ----- 型番の検出 -----
    // "Model:", "型番:", "TYPE:", "製品名:" の後ろ
    if (!result.model) {
      let m = line.match(/(?:MODEL|型番|TYPE|PRODUCT|製品名|品名|品番)[:\s：]*(.+)/i);
      if (m) { result.model = m[1].trim(); continue; }
    }

    // ThinkPad, Latitude, HP, SHARP 等の既知パターン
    if (!result.model) {
      let m = line.match(/(ThinkPad\s+\S+|Latitude\s+\S+|ProBook\s+\S+|EliteBook\s+\S+|MacBook\s+\S+|iPad\s+\S+|MX-\S+|EB-\S+|TM-\S+)/i);
      if (m) { result.model = m[1].trim(); continue; }
    }

    // 汎用: 英数字+ハイフンで6文字以上（型番っぽいもの）
    if (!result.model) {
      let m = line.match(/\b([A-Z]{1,5}[\-\s]?\d{2,}[A-Z0-9\-]*)\b/);
      if (m && m[1].length >= 5) { result.model = m[1].trim(); }
    }

    // ----- シリアル番号の検出 -----
    if (!result.serial) {
      let m = line.match(/(?:S\/N|SERIAL|シリアル|S\.N\.|SN)[:\s：#]*([A-Z0-9\-]{5,})/i);
      if (m) { result.serial = m[1].trim(); continue; }
    }

    // 長い英数字列（10文字以上）をシリアルっぽいと判定
    if (!result.serial) {
      let m = line.match(/\b([A-Z0-9]{10,})\b/);
      if (m && !result.model.includes(m[1])) {
        result.serial = m[1].trim();
      }
    }
  }

  return result;
}

// ---------- テスト用: 手動で1行だけOCR ----------
function testOcrRow2() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CONFIG.sheetName);
  const url = sheet.getRange(2, CONFIG.col.photoUrl).getValue();
  if (!url) {
    Logger.log('N2セルに写真URLを入れてから実行してください');
    return;
  }
  const result = ocrFromDriveUrl(url);
  Logger.log('=== OCR生テキスト ===');
  Logger.log(result.text);
  Logger.log('=== パース結果 ===');
  Logger.log(JSON.stringify(parseOcrText(result.text), null, 2));
}

// ---------- メニュー追加 ----------
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('🦉 備品管理')
    .addItem('📸 OCR一括実行', 'runOcrAll')
    .addItem('🧪 テスト (2行目のみ)', 'testOcrRow2')
    .addToUi();
}

Step 3: スプレッドシートに補助列を追加

GASが結果を書き込むために、スプレッドシートに2列追加します。

列ヘッダー名用途
Y列（25列目）OCR生テキストCloud Visionが読み取った全文（デバッグ用）
Z列（26列目）OCRステータス処理状況（✅完了 / ⚠️検出なし / ❌エラー）
OCR生テキスト	OCRステータス

↑ をY1セルに貼り付けてください。

💡 OCR生テキスト列はデバッグ用。パース精度を確認し終わったら非表示にしてOK。

Step 6: 本番運用の設定

6-1. スプレッドシートにメニュー追加

GASの onOpen() により、スプレッドシートを開くと自動で「🦉 備品管理」メニューが追加されます。

メニュー項目動作
📸 OCR一括実行全行スキャン。写真URLがありOCR未実行の行だけ処理

6-2. 自動トリガー設定（オプション）

写真が追加されるたびに自動でOCRを走らせたい場合：

  1. GASエディタ左メニューの時計アイコン（トリガー）をクリック
  2. 右下「+ トリガーを追加」
  3. 関数: runOcrAll
  4. イベントのソース: 時間主導型
  5. 時間ベースのトリガータイプ: 分ベースのタイマー
  6. 間隔: 5分おき（または15分おき）
  7. 保存
💡 5分おきなら、写真を撮った後少し待てば自動で型番が転記されます。棚卸し期間中だけトリガーをONにし、終わったらOFFにすると無駄なAPI呼び出しが減ります。

6-3. 安全設計のポイント

機能説明
上書き防止型番・シリアル列に既に値がある場合は上書きしない。手入力を優先
再実行しないOCRステータス列に値がある行はスキップ。二重課金なし
エラー記録失敗時はステータスに ❌ + エラー内容を記録。ログ追跡可能

パースロジックのカスタマイズ

parseOcrText() 関数が型番・シリアルを推定するロジックです。精度テストの結果に応じてパターンを追加できます。

例: 大学独自の管理番号パターンを追加

管理番号が J-XXXX 形式なら、以下を parseOcrText() 内に追加：

// 大学管理番号 (J-0001 形式)
if (!result.assetId) {
  let m = line.match(/\b(J-\d{4,})\b/i);
  if (m) { result.assetId = m[1].trim(); }
}

取得した result.assetId をシートの管理番号列に転記する処理を runOcrAll() に追加してください。

💡 テスト5枚のOCR生テキストを見れば、実際にどんな文字列が読み取られるかが分かります。それを元にパターンを調整するのが最も確実。
